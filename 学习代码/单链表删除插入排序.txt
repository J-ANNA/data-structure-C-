//单链表的删除
node *del(node *head,int num)
{
	node *p1,*p2;
	p1 = head;
	while(num != p1->data && p1->next != NULL)   //找到指定删除点
	{p2 = p1; p1 = p1->next;}
	if(num == p1->data)
	{
		if(p1 == head)                         //删除头结点
		{
			head = p1->next;
			free(p1);
		}
		else
		p2->next = p1->next;                 //删除中间节点
	}
	else
		printf("\n%d could not been foud",num);   //为查询到指定节点
		return(head);
}
